  """
  Pads sequences to the specified length.

  Args:
    sequences: Sequences to pad.
    maxlen: Maximum length of sequences.

    truncating: Strategy for truncating.
    value: Value to pad with.

  Returns:
    Padded sequences.
  """

  padded_sequences = []
  for sequence in sequences:
    padded_sequence = sequence[:maxlen]
    if len(sequence) < maxlen:
      if padding == 'pre':
        padded_sequence = [value] * (maxlen - len(sequence)) + padded_sequence

    padded_sequences.append(padded_sequence)
  return padded_sequences
